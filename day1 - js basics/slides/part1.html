Truthy / falsy

En javascript on peut utiliser n'importe quel type de valeur (string, number, ...) comme si c'était un Boolean.

if("not a boolean"){
	console.log('string is truthy');
}else{
	console.log('string is falsy');
}

Calculateur de stock de bananes

Combien de bananes ais-je besoin pour pouvoir en manger
tous les jour jusqu'à la fin de mes jours ?


  • Stocker mon âge dans une variable
  • Stocker mon âge maximum dans une variable
  • Stocker la quantité de bananes mangées par jours dans une variable
  • Calculer la quantité nécessaire pour tenir jusqu'à l'âge maximum
  • Afficher le texte: "Vous aurez besoin de NN bananes pour tenir jusq'à l'âge vénérable de NN ans."

var myAge = 26;
var myMaxAge = 85;
var amountOfBananaPerDay = 6;

var remainingYears = myMaxAge - myAge;
var remainingDays = remainingYears * 365;
var amountNeeded = amountOfBananaPerDay * remainingDays;

console.log("Vous aurez besoin de "+amountNeeded+" bananes pour tenir jusq'à l'âge de "+myMaxAge+" ans.");

if/else


On peut utiliser else pour spécifier une alternative:


	var bananas = 2;
	if(bananas < 3){
	    console.log("Buy more bananas!");
	}else{
	    console.log("You have enough bananas.");
	}

if/else


On peut utiliser else if pour spécifier plusieurs alternatives:



	if(bananas > 10){
	    console.log("You have too many bananas.");
	}else if(bananas < 3){
	    console.log("Buy more bananas!")
	}else if(bananas === 1){
	    console.log("It is the last banana.");
	}else if(bananas === 0){
	    console.log("No bananas :(");
	}else{
	    console.log("Nagative bananas again... How do you do that ?");
	}

Fonctions


Une fonction n'est qu'une liste d'instructions.


Déclarer une fonction:


	function sayHello(){
	    console.log("Hello!");
	}

Appeler une fonction:


	sayHello();

(on dit aussi "exécuter" ou "invoquer" une fonction)

Arguments

Une fonction peut accepter des arguments.


function sayHello(name){
    console.log("Hello " + name + " !");
}

sayHello("Walter White");
sayHello("Jess Pinkman");

var aName = "Skyler White";
sayHello(aName);

Arguments

Une fonction peut accepter PLUSIEURS arguments.


function multiplyNumbers(num1, num2){
    var result = num1 * num2;
    console.log(result);
}

multiplyNumbers(2, 10);

var number1 = 3;
var number2 = 2;

multiplyNumbers(number1, number2);

Return value

Une fonction peut returner une valeur.


function multiplyNumbers(num1, num2){
    var result = num1 * num2;
    return result; // return est la dernière ligne executée par une fonction
}

var number1 = 2;
var number2 = 3;
var total = multiplyNumbers(number1, number2);

console.log(number1+" * "+number2+" = "+total);

Variable scope

Une variable n'est visible que dans la fonction dans laquelle elle est définie: variable locale


function multiplyNumbers(num1, num2){
	// la variable result est "locale"
    var result = num1 * num2;
    return result;
}

var number1 = 2;
var number2 = 3;
var total = multiplyNumbers(number1, number2);

console.log("result:" + result);

Variable scope

Une variable n'est visible que dans la fonction dans laquelle elle est définie: variable globale


// la variable result est "globale"
var result;
function multiplyNumbers(num1, num2){
	// on change la valeur de la variable globale
	// plutot que de déclarer une variable locale
    result = num1 * num2;
    return result;
}

var number1 = 2;
var number2 = 3;
var total = multiplyNumbers(number1, number2);

console.log("result:" + result);
